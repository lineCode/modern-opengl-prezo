// Vertex Shader
varying vec4 ecPosition;
varying vec2 tc;
varying vec3 norm;
void main() {
    ecPosition = gl_ModelViewMatrix * gl_Vertex;
    norm = gl_NormalMatrix * gl_Normal;
    tc = gl_MultiTexCoord0.st;
    gl_Position = ftransform();
}
// Fragment Shader
uniform sampler2D Baboon;
varying vec2 tc;
varying vec3 norm;
void main() {
	float df = dot(vec3(0,0,1),norm);
    vec3 c = df * texture2D(Baboon, tc).rgb;
    gl_FragColor = vec4(c, 1);
}
